// База данных: Производство.xlsx, Цены.xlsx, Расчет стоимости.xlsx, Заказ покупателя.xlsx, Спецификация.xlsx


Table Product {
  product_id int [pk, increment]
  product_code varchar [unique]
  product_name varchar [not null]
  unit_of_measure varchar
  sale_price decimal
  created_at timestamp
}

Table Material {
  material_id int [pk, increment]
  material_code varchar [unique]
  material_name varchar [not null]
  unit_of_measure varchar
  purchase_price decimal
  created_at timestamp
}

Table Production {
  production_id int [pk, increment]
  production_number varchar [unique]
  production_date date [not null]
  product_id int 
  quantity_produced decimal [not null]
  total_cost decimal
  status varchar
}

Table ProductionMaterial {
  id int [pk, increment]
  production_id int 
  material_id int 
  quantity_used decimal [not null]
  unit_price_at_time decimal
  line_total decimal
  
}

Table Customer {
  customer_id int [pk, increment]
  customer_name varchar [not null]
  inn varchar
  contact_info varchar
}

Table Manufacturer {
  manufacturer_id int [pk, increment]
  manufacturer_name varchar [not null]
  address varchar
}

Table CustomerOrder {
  order_id int [pk, increment]
  order_number varchar [unique]
  order_date date [not null]
  customer_id int 
  manufacturer_id int 
  total_amount decimal
  status varchar
}

Table OrderItem {
  order_item_id int [pk, increment]
  order_id int 
  product_id int 
  quantity decimal [not null]
  unit_price decimal [not null]
  line_total decimal
}

Table Specification {
  specification_id int [pk, increment]
  specification_name varchar [not null]
  product_id int
  manufacturer_id int 
  version varchar
  is_active boolean [default: true]
}

Table SpecificationMaterial {
  id int [pk, increment]
  specification_id int 
  material_id int 
  quantity_required decimal [not null]
  unit_of_measure varchar
  
}

Table PriceHistory {
  price_id int [pk, increment]
  item_id int
  item_type enum('product', 'material')
  price decimal [not null] 
  effective_from date
  effective_to date
  created_at timestamp
}


// Связи между сущностями


Ref: Production.product_id > Product.product_id
Ref: ProductionMaterial.production_id > Production.production_id
Ref: ProductionMaterial.material_id > Material.material_id
Ref: CustomerOrder.customer_id > Customer.customer_id
Ref: CustomerOrder.manufacturer_id > Manufacturer.manufacturer_id
Ref: OrderItem.order_id > CustomerOrder.order_id
Ref: OrderItem.product_id > Product.product_id
Ref: Specification.product_id > Product.product_id
Ref: Specification.manufacturer_id > Manufacturer.manufacturer_id
Ref: SpecificationMaterial.specification_id > Specification.specification_id
Ref: SpecificationMaterial.material_id > Material.material_id
