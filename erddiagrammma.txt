Table Организация {
  id integer [primary key]
  название varchar [not null]
  адрес varchar
  телефон varchar
  
  note: 'Организация (ООО Молочный комбинат "Полесье")'
}

Table Заказчик {
  id varchar [primary key]
  название varchar [not null]
  ИНН varchar
  адрес varchar
  телефон varchar
  является_поставщиком boolean [default: false]
  является_покупателем boolean [default: false]
  
  indexes {
    ИНН [unique]
    название
  }
  note: 'Контрагенты из Заказчики.json'
}

Table Продукция {
  код varchar [primary key]
  наименование varchar [not null]
  единица_измерения varchar [not null]
  
  indexes {
    наименование
  }
  note: 'Готовая продукция (из Производство.xlsx)'
}

Table Материал {
  код varchar [primary key]
  наименование varchar [not null]
  единица_измерения varchar [not null]
  
  indexes {
    наименование
  }
  note: 'Сырье и материалы (из Производство.xlsx)'
}

Table Спецификация {
  id integer [primary key]
  наименование varchar [not null]
  код_продукции varchar [ref: > Продукция.код]
  количество_продукции decimal [not null]
  единица_измерения_продукции varchar [not null]
  активна boolean [default: true]
  
  indexes {
    код_продукции
    наименование
  }
  note: 'Технологические карты из Спецификация.xlsx'
}

Table Состав_спецификации {
  id integer [primary key]
  id_спецификации integer [ref: > Спецификация.id]
  код_материала varchar [ref: > Материал.код]
  количество_материала decimal [not null]
  единица_измерения_материала varchar [not null]
  
  indexes {
    (id_спецификации, код_материала) [unique]
  }
  note: 'Строки спецификации из Спецификация.xlsx'
}
